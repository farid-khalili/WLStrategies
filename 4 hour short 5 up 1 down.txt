using System;
using System.Collections.Generic;
using WealthLab.Backtest;
using WealthLab.Core;
using WealthLab.Indicators;

namespace WealthScript7 
{
	public class Short_15Min_5DayRun_FirstRedDay_NoAdx_Hold4Hours : UserStrategyBase
	{
		// ===== USER STANDARD LIQUIDITY FILTERS =====
		private const double MinPrice = 5.0;
		private const int DollarVolLookback = 20;
		private const double MinAvgDollarVol = 500000.0;
		private const double MinAvgVolume20 = 400000.0;

		// ===== SETUP =====
		private const int RunUpDays = 5;
		private const int MinGreenDaysInRun = 4;   // set to 5 for strict "5 straight green days"
		private const int AtrLength = 14;

		// Total run-up across the 5-day run:
		// from OPEN of first run day to CLOSE of fifth run day
		private const double MinRunUpPct = 8.0;
		private const double MaxRunUpPct = 40.0;

		// ===== EXITS =====
		// 4 hours on 15-minute bars = 16 bars
		// Because OrderType.Market exits on the NEXT bar,
		// trigger after 15 elapsed bars from EntryBar.
		private const int HoldBars = 16;

		// ===== REGULAR SESSION ENTRY WINDOW =====
		// 15-minute bars timestamped by BAR CLOSE:
		// 9:45 = first RTH bar, 16:00 = last RTH bar
		private static readonly TimeSpan EntryStartTime = new TimeSpan(9, 45, 0);
		private static readonly TimeSpan EntryEndTime = new TimeSpan(16, 0, 0);

		// ===== ALIGNED SERIES =====
		private bool[] _entrySignal;
		private double[] _avgDollarVol20;
		private double[] _avgVolume20;
		private double[] _yesterdayMoveAtr;
		private double[] _priorDayMoveAtr;
		private double[] _runUpPct5Days;

		public override void Initialize(BarHistory bars)
		{
			StartIndex = 1;
			BuildDailyAlignedData(bars);
		}

		public override void Execute(BarHistory bars, int idx)
		{
			TimeSpan t = bars.DateTimes[idx].TimeOfDay;

			// ===== EXIT LOGIC =====
			if (HasOpenPosition(bars, PositionType.Short))
			{
				Position p = LastOpenPosition;
				if (p != null)
				{
					if (idx - p.EntryBar >= HoldBars - 1)
						ClosePosition(p, OrderType.Market);
				}
				return;
			}

			// ===== ENTRY LOGIC =====
			if (!IsAllowedShortEntryBar(t))
				return;

			// Daily signal is aligned only to the first regular-session bar of the date
			if (_entrySignal[idx])
				PlaceTrade(bars, TransactionType.Short, OrderType.Market);
		}

		private void BuildDailyAlignedData(BarHistory bars)
		{
			int count = bars.Count;

			_entrySignal = new bool[count];
			_avgDollarVol20 = new double[count];
			_avgVolume20 = new double[count];
			_yesterdayMoveAtr = new double[count];
			_priorDayMoveAtr = new double[count];
			_runUpPct5Days = new double[count];

			for (int i = 0; i < count; i++)
			{
				_avgDollarVol20[i] = double.NaN;
				_avgVolume20[i] = double.NaN;
				_yesterdayMoveAtr[i] = double.NaN;
				_priorDayMoveAtr[i] = double.NaN;
				_runUpPct5Days[i] = double.NaN;
			}

			if (count == 0)
				return;

			// Build date -> bar ranges and date -> first regular-session bar
			Dictionary<DateTime, int> firstBarByDate = new Dictionary<DateTime, int>();
			Dictionary<DateTime, int> lastBarByDate = new Dictionary<DateTime, int>();
			Dictionary<DateTime, int> firstRthBarByDate = new Dictionary<DateTime, int>();

			for (int i = 0; i < count; i++)
			{
				DateTime dt = bars.DateTimes[i];
				DateTime date = dt.Date;
				TimeSpan t = dt.TimeOfDay;

				if (!firstBarByDate.ContainsKey(date))
					firstBarByDate[date] = i;

				lastBarByDate[date] = i;

				if (!firstRthBarByDate.ContainsKey(date) && IsAllowedShortEntryBar(t))
					firstRthBarByDate[date] = i;
			}

			BarHistory dailyBars = BarHistoryCompressor.ToDaily(bars);
			if (dailyBars == null || dailyBars.Count == 0)
				return;

			TimeSeries atr14 = ATR.Series(dailyBars, AtrLength);
			int n = dailyBars.Count;

			// Precompute 20-day averages using COMPLETED daily bars before the entry date.
			double[] avgDollarVol20Daily = new double[n];
			double[] avgVolume20Daily = new double[n];

			for (int d = 0; d < n; d++)
			{
				avgDollarVol20Daily[d] = double.NaN;
				avgVolume20Daily[d] = double.NaN;

				if (d >= DollarVolLookback)
				{
					double sumDollarVol = 0.0;
					double sumVolume = 0.0;

					for (int k = d - DollarVolLookback; k < d; k++)
					{
						sumDollarVol += dailyBars.Close[k] * dailyBars.Volume[k];
						sumVolume += dailyBars.Volume[k];
					}

					avgDollarVol20Daily[d] = sumDollarVol / DollarVolLookback;
					avgVolume20Daily[d] = sumVolume / DollarVolLookback;
				}
			}

			// For each DAILY bar date d (= entry date), use d-1 as the last completed daily bar.
			for (int d = 0; d < n; d++)
			{
				DateTime entryDate = dailyBars.DateTimes[d].Date;

				if (!firstBarByDate.ContainsKey(entryDate) || !lastBarByDate.ContainsKey(entryDate))
					continue;

				int fillFrom = firstBarByDate[entryDate];
				int fillTo = lastBarByDate[entryDate];

				double avgDollarVol20 = avgDollarVol20Daily[d];
				double avgVolume20 = avgVolume20Daily[d];

				bool priorRunQualified = false;
				bool yesterdayWasRed = false;
				bool atrContraction = false;

				double runUpPct = double.NaN;
				double yesterdayMoveAtr = double.NaN;
				double priorDayMoveAtr = double.NaN;

				// Need:
				// d-6 .. d-2 = 5-day run
				// d-1        = yesterday, the red day
				// d          = today, entry day
				if (d >= RunUpDays + 1)
				{
					int runDay1 = d - 6;
					int runDay2 = d - 5;
					int runDay3 = d - 4;
					int runDay4 = d - 3;
					int runDay5 = d - 2;
					int yesterday = d - 1;

					int greenCount = 0;
					if (dailyBars.Close[runDay1] > dailyBars.Open[runDay1]) greenCount++;
					if (dailyBars.Close[runDay2] > dailyBars.Open[runDay2]) greenCount++;
					if (dailyBars.Close[runDay3] > dailyBars.Open[runDay3]) greenCount++;
					if (dailyBars.Close[runDay4] > dailyBars.Open[runDay4]) greenCount++;
					if (dailyBars.Close[runDay5] > dailyBars.Open[runDay5]) greenCount++;

					priorRunQualified = greenCount >= MinGreenDaysInRun;
					yesterdayWasRed = dailyBars.Close[yesterday] < dailyBars.Open[yesterday];

					// 5-day run-up measured from open of first run day
					// to close of fifth run day (the day before yesterday)
					double firstOpen = dailyBars.Open[runDay1];
					if (firstOpen > 0.0)
						runUpPct = 100.0 * (dailyBars.Close[runDay5] - firstOpen) / firstOpen;

					// Compare yesterday's ATR-normalized close-to-close move
					// vs the prior day's ATR-normalized close-to-close move.
					if (d >= 3 &&
						!double.IsNaN(atr14[d - 1]) && atr14[d - 1] > 0.0 &&
						!double.IsNaN(atr14[d - 2]) && atr14[d - 2] > 0.0)
					{
						yesterdayMoveAtr = Math.Abs(dailyBars.Close[d - 1] - dailyBars.Close[d - 2]) / atr14[d - 1];
						priorDayMoveAtr = Math.Abs(dailyBars.Close[d - 2] - dailyBars.Close[d - 3]) / atr14[d - 2];

						atrContraction = yesterdayMoveAtr < priorDayMoveAtr;
					}
				}

				bool passesLiquidity =
					d >= DollarVolLookback &&
					dailyBars.Close[d - 1] >= MinPrice &&
					avgDollarVol20 >= MinAvgDollarVol &&
					avgVolume20 >= MinAvgVolume20;

				bool passesRunUp =
					!double.IsNaN(runUpPct) &&
					runUpPct >= MinRunUpPct &&
					runUpPct <= MaxRunUpPct;

				bool signal =
					priorRunQualified &&
					yesterdayWasRed &&
					passesRunUp &&
					atrContraction &&
					passesLiquidity;

				// Fill aligned series for all bars on this calendar date
				for (int i = fillFrom; i <= fillTo; i++)
				{
					_avgDollarVol20[i] = avgDollarVol20;
					_avgVolume20[i] = avgVolume20;
					_yesterdayMoveAtr[i] = yesterdayMoveAtr;
					_priorDayMoveAtr[i] = priorDayMoveAtr;
					_runUpPct5Days[i] = runUpPct;
				}

				// Only allow entry on the first regular-session bar of the entry date
				if (signal && firstRthBarByDate.TryGetValue(entryDate, out int firstRthBar))
					_entrySignal[firstRthBar] = true;
			}
		}

		private bool IsAllowedShortEntryBar(TimeSpan t)
		{
			return t >= EntryStartTime && t <= EntryEndTime;
		}
	}
}
